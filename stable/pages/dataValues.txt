function scrollIt() {
	numEntries = $("#" + getTab(selectedTab) + " span").length;
	prevScrollRow = scrollRow;
	scrollRow = Math.floor(($("#scrollHandle").position().top - $("#scrollBar").position().top) / (($("#scrollBar").height() - 29) / Math.ceil((numEntries + 1) / 9 - 4)));
	if(prevScrollRow != scrollRow) updateEntries();
}

function setSelected(id) {
	tabSrc = $("#tab_" + selectedTab).attr("src");
	$("#tab_" + selectedTab).attr("src", tabSrc.substring(0, tabSrc.length - 12) + ".png").css("z-index", "0");
	selectedTab = Number(id.substring(id.lastIndexOf('_') + 1));
	tabSrc =  $("#" + id).attr("src");
	$("#" + id).attr("src", tabSrc.substring(0, tabSrc.length - 4) + "Selected.png").css("z-index", "2");
	$("#tabTitle").html($("#" + getTab(selectedTab)).attr("title"));
	scrollRow = 0;
	$("#scrollHandle").css({ top: dWidgetPos.top + 92, left: dWidgetPos.left + 350 });
	updateEntries();
}

function updateEntries() {
	imageSlots = $(".entryItem");
	tabEntries = $("#" + getTab(selectedTab) + " span");
	for(i=0;i<imageSlots.length - 9;i++) {
		slotAt = i + scrollRow * 9;
		if(tabEntries[slotAt] == null) {
			$(imageSlots[i]).attr("src", "resources/images/dataWidget/blankSlot.png");
			entryLinks[i] = null;
			continue;
		}
		$(imageSlots[i]).attr("src", "resources/images/" + $(tabEntries[slotAt]).attr("type") + "s/" + ($(tabEntries[slotAt]).attr("src") == null ? $(tabEntries[slotAt]).attr("id").substring(6) + ".png" : $(tabEntries[slotAt]).attr("src")));
		entryLinks[i] = $(tabEntries[slotAt]).attr("id");
	}
	if(tabEntries.length < 45) {
		$("#scrollHandle").attr("disabled", "disabled").attr("src", "resources/images/dataWidget/scrollHandleDisabled.png");
	}
	else {
		$("#scrollHandle").removeAttr("disabled").attr("src", "resources/images/dataWidget/scrollHandle.png");
	}
}

function getTab(selTab) {
	return selTab == 5 ? "tabAllSearch" : selTab == 11 ? "tabInventory" : pages[pageAt][selTab - (selTab > 5 ? 1 : 0)];
}
